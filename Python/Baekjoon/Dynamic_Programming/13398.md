```
import sys
inputs = sys.stdin.readline

N = int(inputs())
A = list(map(int, inputs().split()))
dp = [[x for x in A] for _ in range(2)]

for i in range(1,N):
    dp[0][i] = max(dp[0][i-1] + A[i], dp[0][i])
    dp[1][i] = max(dp[0][i-1], dp[1][i-1] + A[i])
    
print(max(max(dp[0]), max(dp[1])))
```
