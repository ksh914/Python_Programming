    dp[j][1] = dp[j-2][0]%1000000009 + dp[j-2][2]%1000000009
    dp[j][2] = dp[j-3][0]%1000000009 + dp[j-3][1]%1000000009
    
for k in range(T):

    print(sum(dp[n])%1000000009)
