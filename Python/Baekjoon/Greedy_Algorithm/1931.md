import sys
N = int(sys.stdin.readline())
ML=[]
for i in range(N):
    ML.append(list(map(int,sys.stdin.readline().split())))
    
ML.sort(key=lambda x:x[0])
ML.sort(key=lambda x:x[1])

cnt=0
comp=0
for A,B in ML:
    if A>=comp:
        cnt+=1
        comp=B
        
print(cnt)
